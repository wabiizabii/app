import streamlit as st
from core.mt5_handler import MT5Handler
from core.supabase_handler import SupabaseHandler

@st.cache_resource
def get_mt5_handler():
    return MT5Handler()

def render(db_handler: SupabaseHandler) -> bool:
    st.sidebar.title("⚙️ Control Panel")
    st.sidebar.markdown("---")

    # --- Portfolio Selector ---
    st.sidebar.subheader("🎯 Portfolio Selector")
    df_portfolios = db_handler.load_portfolios()
    
    if df_portfolios.empty:
        st.sidebar.warning("No portfolios found.")
        st.session_state['active_portfolio_id'] = None
        return True # Return True to avoid blocking the user from creating a portfolio

    portfolio_options = dict(zip(df_portfolios['PortfolioName'], df_portfolios['PortfolioID']))
    
    def on_portfolio_change():
        selected_name = st.session_state.portfolio_selector
        st.session_state.active_portfolio_name = selected_name
        st.session_state.active_portfolio_id = portfolio_options.get(selected_name)

    # Determine the index for the selectbox
    portfolio_names = list(portfolio_options.keys())
    try:
        current_index = portfolio_names.index(st.session_state.get('active_portfolio_name'))
    except (ValueError, TypeError):
        current_index = 0
        if portfolio_names: # Set the first portfolio as default if none is selected
            st.session_state.active_portfolio_name = portfolio_names[0]
            st.session_state.active_portfolio_id = portfolio_options[portfolio_names[0]]

    st.sidebar.selectbox(
        "Select Portfolio:",
        options=portfolio_names,
        index=current_index,
        key='portfolio_selector',
        on_change=on_portfolio_change
    )

    # --- MT5 Connection ---
    st.sidebar.markdown("---")
    if st.sidebar.button("🔄 Refresh MT5 Data", use_container_width=True, type="primary"):
        st.rerun()

    mt5_handler = get_mt5_handler()
    if not mt5_handler.initialize_connection():
        st.sidebar.error("❌ MT5 not connected.")
        st.session_state['live_mt5_account'] = None
        return False

    account_info = mt5_handler.get_account_info()
    if not account_info:
        st.sidebar.error("❌ Failed to get MT5 account info.")
        st.session_state['live_mt5_account'] = None
        return False
    
    live_mt5_id = str(account_info.get('login'))
    st.session_state['live_mt5_account'] = live_mt5_id
    st.sidebar.success(f"Connected to MT5: {live_mt5_id}")
    
    # --- Link Validation (Guardrail) ---
    is_link_valid = False
    active_portfolio_id = st.session_state.get('active_portfolio_id')
    
    if active_portfolio_id:
        portfolio_details = db_handler.get_portfolio_details(active_portfolio_id)
        if portfolio_details:
            linked_mt5_id_in_db = portfolio_details.get('mt5_account_id')
            st.session_state['linked_mt5_id'] = linked_mt5_id_in_db
            
            if linked_mt5_id_in_db is None or str(linked_mt5_id_in_db) == live_mt5_id:
                is_link_valid = True
            else:
                st.sidebar.error(f"Mismatch: Portfolio expects {linked_mt5_id_in_db}.")
                is_link_valid = False
        else:
            is_link_valid = False # Should not happen if portfolio is selected
    
    # --- Trading Setup Expander ---
    st.sidebar.markdown("---")
    with st.sidebar.expander("⚙️ Trading Setup"):
        st.info("Trade planning settings will be shown here.")
        # Your previous Trading Setup UI (FIBO/CUSTOM) can be placed here.
        
    return is_link_valid
