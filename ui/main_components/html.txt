    <script>
        function openTab(evt, tabName) {
            let i, tabcontent, tablinks;
            tabcontent = document.getElementsByClassName("tab-content");
            for (i = 0; i < tabcontent.length; i++) { tabcontent[i].style.display = "none"; }
            tablinks = document.getElementsByClassName("tab-button");
            for (i = 0; i < tablinks.length; i++) { tablinks[i].className = tablinks[i].className.replace(" active", ""); }
            document.getElementById(tabName).style.display = "block";
            evt.currentTarget.className += " active";
        }
        
        const ws = new WebSocket("ws://localhost:5555");
        
        function formatCurrency(v, d=2) { return '$' + (v || 0).toLocaleString('en-US', { minimumFractionDigits:d, maximumFractionDigits:d }); }
        
        function updateProgressBar(id, val, min, max) {
            if (!id || val === undefined || min === undefined || max === undefined) return;
            const range = max - min;
            let pct = (range > 0) ? ((val - min) / range) * 100 : (val >= max ? 100 : 0);
            const thumb = document.getElementById(id);
            if (thumb) thumb.style.left = `${Math.max(0, Math.min(100, pct))}%`;
        }

        function showNotification({ type = 'info', title, message, duration = 7000 }) {
            const container = document.getElementById('notification-container');
            const icons = { success: '✅', error: '❌', info: '🔔' };
            const toast = document.createElement('div');
            toast.className = `notification-toast ${type}`;
            toast.innerHTML = `<div class="icon">${icons[type] || ''}</div><div class="content"><div class="title">${title}</div><div class="message">${message}</div></div>`;
            container.appendChild(toast);
            setTimeout(() => {
                toast.classList.add('fade-out');
                toast.addEventListener('transitionend', () => toast.remove());
            }, duration);
        }

        function setupUploadFunctionality() {
            const uploadButton = document.getElementById('upload-button');
            const portfolioSelect = document.getElementById('portfolio-select-for-upload');
            const fileInput = document.getElementById('statement-file-input');
            const uploadStatus = document.getElementById('upload-status');

            if (uploadButton && !uploadButton.listenerAttached) {
                uploadButton.addEventListener('click', async () => {
                    const file = fileInput.files[0];
                    if (!portfolioSelect.value || !file) {
                        uploadStatus.textContent = '❌ Please select portfolio and file.';
                        uploadStatus.style.color = 'var(--color-red)';
                        return;
                    }
                    uploadStatus.textContent = 'Uploading...';
                    const formData = new FormData();
                    formData.append('statementFile', file);
                    formData.append('portfolioName', portfolioSelect.value);
                    try {
                        const response = await fetch('http://localhost:5001/upload', { method: 'POST', body: formData });
                        const result = await response.json();
                        uploadStatus.textContent = response.ok ? `✅ ${result.message}` : `❌ ${result.message}`;
                        uploadStatus.style.color = response.ok ? 'var(--color-green)' : 'var(--color-red)';
                    } catch (error) {
                        uploadStatus.textContent = '❌ Network Error.';
                        uploadStatus.style.color = 'var(--color-red)';
                    }
                });
                uploadButton.listenerAttached = true;
            }
        }

        ws.onopen = () => { document.getElementById('connection-status').textContent = '✅ Connected. Waiting for data...'; };
        ws.onclose = () => { document.getElementById('connection-status').innerHTML = '<span style="color:red;">❌ Disconnected. Restart service.</span>'; document.getElementById('main-content').style.display = 'none';};
        ws.onerror = (err) => { document.getElementById('connection-status').innerHTML = '<span style="color:red;">❌ Connection Error. Is backend running?</span>'; document.getElementById('main-content').style.display = 'none';};

        ws.onmessage = (event) => {
            try {
                if (document.getElementById('main-content').style.display === 'none') {
                    document.getElementById('connection-status').style.display = 'none';
                    document.getElementById('main-content').style.display = 'block';
                }
                const data = JSON.parse(event.data);
                if (!data || !data.account_id) return;

                const ctx = data.portfolio_context;
                const stats = data.performance_analytics;
                
                document.getElementById('portfolio-title').textContent = `#${data.account_id} - ${ctx.portfolio_name || 'Unlinked'}`;
                document.getElementById('portfolio-status').textContent = ctx.portfolio_name ? 'Active' : 'Unlinked';

                const openPositions = data.open_positions || [];
                document.getElementById('positions-total').textContent = `${openPositions.length} Open`;
                document.getElementById('positions-summary').textContent = `${openPositions.filter(p=>p.Profit>0).length}W / ${openPositions.filter(p=>p.Profit<0).length}L`;

                if (ctx && ctx.initial_balance > 0) {
                    const profitTarget = ctx.initial_balance * (ctx.profit_target_pct / 100);
                    const currentProfit = data.daily_net_pl;
                    const stopoutLevel = ctx.initial_balance * (1 - (ctx.total_stopout_pct / 100));
                    document.getElementById('profit-value').textContent = `${formatCurrency(currentProfit)} / ${formatCurrency(profitTarget)}`;
                    document.getElementById('profit-stopout-label').textContent = `Stopout: ${formatCurrency(stopoutLevel)}`;
                    document.getElementById('profit-target-label').textContent = `Target: ${formatCurrency(ctx.initial_balance + profitTarget)}`;
                    updateProgressBar('profit-thumb', data.equity, stopoutLevel, ctx.initial_balance + profitTarget);
                    const dailyLossLimitEquity = ctx.opening_balance * (1 - (ctx.daily_loss_limit_pct / 100));
                    const lossLeft = data.equity - dailyLossLimitEquity;
                    document.getElementById('loss-left-value').textContent = `${formatCurrency(lossLeft)} left`;
                    document.getElementById('loss-limit-label').textContent = `Limit: ${formatCurrency(dailyLossLimitEquity)}`;
                    document.getElementById('loss-start-label').textContent = `Start: ${formatCurrency(ctx.opening_balance)}`;
                    updateProgressBar('loss-thumb', data.equity, dailyLossLimitEquity, ctx.opening_balance);
                }

                document.getElementById('stat-win-rate').innerHTML = `${(stats.win_rate || 0).toFixed(2)}% <span class="metric-label sub-value">(${stats.winning_trades}W / ${stats.losing_trades}L)</span>`;
                document.getElementById('stat-profit-factor').textContent = stats.profit_factor === 'Infinity' ? '∞' : (stats.profit_factor || 0).toFixed(2);
                document.getElementById('stat-avg-win').textContent = formatCurrency(stats.average_win);
                document.getElementById('stat-avg-loss').textContent = formatCurrency(stats.average_loss);

                document.getElementById('detail-equity').textContent = formatCurrency(data.equity);
                document.getElementById('detail-pl').textContent = formatCurrency(data.floating_pl);
                document.getElementById('detail-balance').textContent = formatCurrency(data.balance);
                document.getElementById('detail-risk').textContent = formatCurrency(data.total_open_risk);
                document.getElementById('detail-rr').textContent = (data.avg_rr || 0).toFixed(2);

                async function calculateLotSize() {
                    planStatusDiv.textContent = 'Calculating...';
                    planStatusDiv.style.color = 'var(--text-color-secondary)';
                    
                    try {
                        const response = await fetch('http://localhost:5001/calculate-lot', {

                            headers: { 'Content-Type': 'application/json' },
                            body: JSON.stringify({
                                balance: currentBalance,
                                risk_percent: parseFloat(planRiskInput.value),
                                symbol: planAssetInput.value,
                                entry_price: parseFloat(planEntryInput.value),
                                sl_price: parseFloat(planSlInput.value),
                                account_type: currentAccountType
                            })
                        });

                        if (response.ok) {
                            planLotOutput.value = result.lot_size;
                            planStatusDiv.textContent = `Risk Amount: ${formatCurrency(result.risk_amount)}`;
